// 内存缓存实现 const requestCache = new Map(); const CACHE_DURATION = 5 * 60 * 1000; // 5分钟缓存 function getCachedResult(cacheKey: string): any | null { const cached = requestCache.get(cacheKey); if (cached && Date.now() - cached.timestamp < CACHE_DURATION) { console.log('✅ 使用缓存结果'); return cached.result; } return null; }
